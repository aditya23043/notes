Data Types

char 1 byte %c
char[] / char* string %s
float 4 bytes (32 bits precision) 6-7 digits %f
double 8 bytes (64 bits precision) 15-16 digits %lf
bool (requires stdbool.h) 1 byte %d
char 1 byte (-128 to 127) %d / %c
unsigned char 1 byte (0 to 255) %d / %c
short int 2 bytes %d
unsigned short int 2 bytes %d
int 4 bytes %d
unsigned int 4 bytes %u
long long int 8 bytes %lld
unsigned long long int 8 bytes %llu

File IO


main.c
				
#include <stdio.h>

int main(int argc, char **argv){

	FILE *fp;
	char string[100];
	char *foo = "suckss";

	fp = fopen("foo.txt", "w+");
	
	if(fp == NULL){
		printf("Error while opening file!");
		return 1;
	}

	// WRITING
	fprintf(fp, "Frontend web dev %s\n", foo);

	// sets the file pointer to the head of the file
	rewind(fp);

	// READING
	while(fgets(string, 100, fp) != NULL){
		printf("%s", string);
	}

	fclose(fp);

	return 0;
}
