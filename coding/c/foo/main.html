Constant Value

  • constant values cannot be changed after declaration
  • naming convention : all letters => uppercase

❯ ./a.out
3.141590

Array Notation Vs Pointer Notation

#include <stdio.h> int main(int argc, char *argv[]) { int a[] = { 1, 2, 3, 4, 5 }; int *b; b = a; printf("%d\n", *b); // prints the first element of the array which is being pointed to by the pointer printf("%d\n", b[1]); // array notation printf("%d\n", *(b+2)); // pointer notation printf("%d\n", *(b+3)); // pointer notation return 0; } 

❯ ./a.out
1
2
3
4

File IO

Important Keywords

FILE Initializes a file pointer
fopen Opens a file whose path is given as the 1st arg and in the mode as given in the 2nd arg Open a file and create a new stream for it. FILE *fopen(const char *__filename,
fprintf write to a file using its file pointer in write / append mode Write formatted output to STREAM. int fprintf(FILE *__stream, const char *__format)
fgets used to retrieve strings (char arrays) from the file pointer's current position Get a newline-terminated string of finite length from STREAM. char *fgets(char *__s, int __n, FILE *__stream)
rewind places the file pointer cursor to the beginning Rewind to the beginning of STREAM. void rewind(FILE *__stream)
fclose terminates the file pointer Close STREAM. int fclose(FILE *__stream)

#include <stdio.h> int main(int argc, char **argv){ FILE *fp; char string[100]; char *foo = "sucks"; fp = fopen("foo.txt", "w+"); if(fp == NULL){ printf("Error while opening file!"); return 1; } // WRITING fprintf(fp, "Frontend web dev %s\n", foo); // sets the file pointer to the head of the file rewind(fp); // READING while(fgets(string, 100, fp) != NULL){ printf("%s", string); } fclose(fp); return 0; } 

❯ ./a.out
Frontend web dev sucks

❯ cat foo.txt
Frontend web dev sucks

Dynamically Allocated Memory

malloc Allocate SIZE bytes of memory. void *malloc(size_t __size)
calloc Allocate NMEMB elements of SIZE bytes each, all initialized to 0. void *calloc(size_t __nmemb, size_t __size)
realloc Re-allocate the previously allocated block in PTR, making the new block SIZE bytes long. void *realloc(void *__ptr, size_t __size)
free Free a block allocated by `malloc', `realloc' or `calloc'. void free(void *__ptr)
#include <stdio.h> #include #include int main(int argc, char *argv[]) { int *a; a = malloc(sizeof(int)*5); a[0] = 1; a[1] = 2; a[2] = 3; a[3] = 4; a[4] = 5; for(int i = 0; i < 5; i++){ printf("%d\n", a[i]); // array notation } free(a); return 0; }
